```swift
import SwiftUI

/// Two-pane layout container with fixed 33/67 split
/// Uses GeometryReader instead of NavigationSplitView to guarantee proportions
struct MainSplitView: View {
    @Binding var selectedConversation: SampleConversation?
    let conversations: [SampleConversation]

    var body: some View {
        GeometryReader { geometry in
            HStack(spacing: 0) {
                // Left pane: 33% width - message list
                GlassCard {
                    MessageListPane(
                        conversations: conversations,
                        selectedConversation: $selectedConversation
                    )
                }
                .frame(width: geometry.size.width * 0.33)

                // Thin divider line
                Rectangle()
                    .fill(Color.white.opacity(0.15))
                    .frame(width: 1)

                // Right pane: remaining width - thread view
                GlassCard {
                    ThreadViewPane(conversation: selectedConversation)
                }
            }
            .padding(12) // Gap between panes and window edge
        }
    }
}

    .frame(width: 1000, height: 600)
}
```

```swift
import SwiftUI

/// Left pane displaying the list of conversations awaiting response
struct MessageListPane: View {
    let conversations: [SampleConversation]
    @Binding var selectedConversation: SampleConversation?

    var body: some View {
        VStack(alignment: .leading, spacing: 0) {
            // Header
            Text("Inbox")
                .font(.title2)
                .fontWeight(.semibold)
                .foregroundStyle(.white)
                .padding(.horizontal, 16)
                .padding(.vertical, 12)

            Divider()
                .background(Color.white.opacity(0.1))

            // Conversation list
            if conversations.isEmpty {
                // Inbox zero state - minimal/nothing per CONTEXT.md
                Spacer()
            } else {
                ScrollView {
                    LazyVStack(spacing: 0) {
                        ForEach(conversations) { conversation in
                            ConversationRow(
                                conversation: conversation,
                                isSelected: selectedConversation?.id == conversation.id
                            )
                            .onTapGesture {
                                selectedConversation = conversation
                            }
                        }
                    }
                }
            }
        }
    }
}

/// Single row in the conversation list
struct ConversationRow: View {
    let conversation: SampleConversation
    let isSelected: Bool

    var body: some View {
        HStack(spacing: 12) {
            // Source icon
            Image(systemName: conversation.source.iconName)
                .font(.system(size: 14))
                .foregroundStyle(.white.opacity(0.6))
                .frame(width: 20)

            // Content
            VStack(alignment: .leading, spacing: 4) {
                HStack {
                    Text(conversation.participant)
                        .font(.subheadline)
                        .fontWeight(.medium)
                        .foregroundStyle(.white)
                        .lineLimit(1)

                    Spacer()

                    Text(conversation.formattedTime)
                        .font(.caption)
                        .foregroundStyle(.white.opacity(0.5))
                }

                Text(conversation.lastMessage)
                    .font(.caption)
                    .foregroundStyle(.white.opacity(0.7))
                    .lineLimit(2)
            }

            // Unread indicator
            if conversation.unreadCount > 0 {
                Circle()
                    .fill(Color.blue)
                    .frame(width: 8, height: 8)
            }
        }
        .padding(.horizontal, 16)
        .padding(.vertical, 12)
        .background(isSelected ? Color.white.opacity(0.1) : Color.clear)
        .contentShape(Rectangle()) // Make entire row tappable
    }
}
```

Create ThreadViewPane.swift - the right pane showing message thread:

```swift
import SwiftUI

/// Right pane displaying the selected conversation's message thread
struct ThreadViewPane: View {
    let conversation: SampleConversation?

    var body: some View {
        if let conversation = conversation {
            VStack(spacing: 0) {
                // Header with participant name
                HStack {
                    Image(systemName: conversation.source.iconName)
                        .foregroundStyle(.white.opacity(0.6))
                    Text(conversation.participant)
                        .font(.headline)
                        .foregroundStyle(.white)
                    Spacer()
                }
                .padding(.horizontal, 16)
                .padding(.vertical, 12)

                Divider()
                    .background(Color.white.opacity(0.1))

                // Messages
                ScrollView {
                    LazyVStack(spacing: 12) {
                        ForEach(conversation.messages) { message in
                            MessageBubble(message: message)
                        }
                    }
                    .padding(16)
                }

                Divider()
                    .background(Color.white.opacity(0.1))

                // Composer placeholder
                HStack {
                    Text("Reply to \(conversation.participant)...")
                        .font(.subheadline)
                        .foregroundStyle(.white.opacity(0.4))
                    Spacer()
                }
                .padding(.horizontal, 16)
                .padding(.vertical, 12)
            }
        } else {
            // Empty state - instruction text per CONTEXT.md
            VStack {
                Spacer()
                Text("Select a conversation to view")
                    .font(.title3)
                    .foregroundStyle(.white.opacity(0.5))
                Spacer()
            }
        }
    }
}

/// Individual message bubble in the thread
struct MessageBubble: View {
    let message: SampleMessage

    var body: some View {
        HStack {
            if message.isFromMe { Spacer() }

            VStack(alignment: message.isFromMe ? .trailing : .leading, spacing: 4) {
                if !message.isFromMe {
                    Text(message.sender)
                        .font(.caption)
                        .foregroundStyle(.white.opacity(0.5))
                }

                Text(message.content)
                    .font(.body)
                    .foregroundStyle(.white)
                    .padding(.horizontal, 12)
                    .padding(.vertical, 8)
                    .background(
                        message.isFromMe
                            ? Color.blue.opacity(0.6)
                            : Color.white.opacity(0.1)
                    )
                    .clipShape(RoundedRectangle(cornerRadius: 16, style: .continuous))

                Text(formatTime(message.timestamp))
                    .font(.caption2)
                    .foregroundStyle(.white.opacity(0.4))
            }

            if !message.isFromMe { Spacer() }
        }
    }

    private func formatTime(_ date: Date) -> String {
        let formatter = DateFormatter()
        formatter.dateFormat = "h:mm a"
        return formatter.string(from: date)
    }
}

}
```
