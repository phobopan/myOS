import SwiftUI

/// Settings panel that slides from the left with sidebar navigation
struct SettingsPanel: View {
    @Binding var isPresented: Bool
    @State private var selectedSection: SettingsSection = .accounts

    var body: some View {
        GlassCard(cornerRadius: 20) {
            HStack(spacing: 0) {
                // Mini sidebar
                VStack(alignment: .leading, spacing: 8) {
                    // Header with close button
                    HStack {
                        Text("Settings")
                            .font(.headline)
                            .foregroundStyle(.white)
                        Spacer()
                        Button(action: { isPresented = false }) {
                            Image(systemName: "xmark.circle.fill")
                                .font(.title3)
                                .foregroundStyle(.white.opacity(0.6))
                        }
                        .buttonStyle(.plain)
                        .help("Close Settings")
                    }
                    .padding(.bottom, 8)

                    // Section buttons
                    ForEach(SettingsSection.allCases) { section in
                        SettingsSidebarButton(
                            section: section,
                            isSelected: selectedSection == section
                        ) {
                            selectedSection = section
                        }
                    }

                    Spacer()
                }
                .frame(width: 140)
                .padding()

                // Divider
                Rectangle()
                    .fill(Color.white.opacity(0.1))
                    .frame(width: 1)

                // Content area
                VStack(alignment: .leading) {
                    Text(selectedSection.rawValue)
                        .font(.title2)
                        .fontWeight(.semibold)
                        .foregroundStyle(.white)
                        .padding(.bottom, 8)

                    // Section-specific content
                    SettingsSectionContent(section: selectedSection)

                    Spacer()
                }
                .frame(maxWidth: .infinity, alignment: .leading)
                .padding()
            }
        }
        .frame(width: 400, height: 500)
    }
}

/// Settings sections available
enum SettingsSection: String, CaseIterable, Identifiable {
    case accounts = "Accounts"
    case notifications = "Notifications"
    case shortcuts = "Shortcuts"
    case appearance = "Appearance"

    var id: String { rawValue }

    var icon: String {
        switch self {
        case .accounts: return "person.crop.circle"
        case .notifications: return "bell"
        case .shortcuts: return "keyboard"
        case .appearance: return "paintbrush"
        }
    }
}

/// Sidebar button for section navigation
struct SettingsSidebarButton: View {
    let section: SettingsSection
    let isSelected: Bool
    let action: () -> Void

    var body: some View {
        Button(action: action) {
            HStack(spacing: 8) {

    }
}

import SwiftUI

struct ContentView: View {
    @State private var selectedConversation: SampleConversation?
    @State private var showSettings = false

    var body: some View {
        ZStack(alignment: .leading) {
            // Main content
            MainSplitView(
                selectedConversation: $selectedConversation,
                conversations: SampleConversation.samples
            )

            // Dimming overlay when settings open
            if showSettings {
                Color.black.opacity(0.4)
                    .ignoresSafeArea()
                    .onTapGesture {
                        withAnimation(.spring(response: 0.3)) {
                            showSettings = false
                        }
                    }
                    .transition(.opacity)
            }

            // Settings panel - slides from left
            SettingsPanel(isPresented: $showSettings)
                .offset(x: showSettings ? 20 : -400)
                .animation(.spring(response: 0.35, dampingFraction: 0.8), value: showSettings)
        }
        .frame(minWidth: 900, minHeight: 600)
        .background(VisualEffectView().ignoresSafeArea())
        .toolbar {
            ToolbarItem(placement: .automatic) {
                Button(action: {
                    withAnimation(.spring(response: 0.35, dampingFraction: 0.8)) {
                        showSettings.toggle()
                    }
                }) {
                    Image(systemName: "gearshape")
                        .foregroundStyle(.white.opacity(0.7))
                }
                .help("Settings")
                .keyboardShortcut(",", modifiers: .command) // Standard macOS settings shortcut
            }
        }
    }
}
