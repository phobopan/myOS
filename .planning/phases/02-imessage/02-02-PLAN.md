```typescript
import { getAuthStatus, askForFullDiskAccess } from 'node-mac-permissions';

export type PermissionStatus = 'authorized' | 'denied' | 'not-determined';

export function checkFullDiskAccess(): PermissionStatus {
  const status = getAuthStatus('full-disk-access');
  if (status === 'authorized') return 'authorized';
  if (status === 'not determined') return 'not-determined';
  return 'denied';
}

export function requestFullDiskAccess(): void {
  // Opens System Preferences > Privacy & Security > Full Disk Access
  // Cannot programmatically grant - user must toggle the checkbox
  askForFullDiskAccess();
}
```

```typescript
import { parsePhoneNumber } from 'libphonenumber-js';

// node-mac-contacts uses CommonJS, import dynamically
let macContacts: typeof import('node-mac-contacts') | null = null;

async function getMacContacts() {
  if (!macContacts) {
    macContacts = await import('node-mac-contacts');
  }
  return macContacts;
}

export interface ContactInfo {
  firstName?: string;
  lastName?: string;
  displayName: string;
}

// Cache for fast lookups
let contactCache: Map<string, ContactInfo> = new Map();
let cacheBuilt = false;

export async function buildContactCache(): Promise<void> {
  const contacts = await getMacContacts();

  const authStatus = contacts.getAuthStatus();
  if (authStatus !== 'Authorized') {
    const result = await contacts.requestAccess();
    if (result !== 'Authorized') {
      console.log('Contacts access not authorized');
      return;
    }
  }

  const allContacts = contacts.getAllContacts();
  const newCache = new Map<string, ContactInfo>();

  for (const contact of allContacts) {
    const displayName = [contact.firstName, contact.lastName]
      .filter(Boolean)
      .join(' ') || contact.nickname || 'Unknown';

    const info: ContactInfo = {
      firstName: contact.firstName,
      lastName: contact.lastName,
      displayName,
    };

    // Index by phone numbers (normalized to E.164)
    if (contact.phoneNumbers) {
      for (const phone of contact.phoneNumbers) {
        try {
          const parsed = parsePhoneNumber(phone, 'US');
          if (parsed) {
            newCache.set(parsed.format('E.164'), info); // +15551234567
          }
        } catch {
          // Fallback to raw number
          newCache.set(phone.replace(/\D/g, ''), info);
        }
      }
    }

    // Index by email addresses (lowercase)
    if (contact.emailAddresses) {
      for (const email of contact.emailAddresses) {
        newCache.set(email.toLowerCase(), info);
      }
    }
  }

  contactCache = newCache;
  cacheBuilt = true;
  console.log(`Contact cache built with ${newCache.size} entries`);
}

export function resolveHandle(handle: string | null): ContactInfo | null {
  if (!handle || !cacheBuilt) return null;

  // Try direct lookup first
  const direct = contactCache.get(handle.toLowerCase());
  if (direct) return direct;

  // Try E.164 normalized phone
  try {
    const parsed = parsePhoneNumber(handle, 'US');
    if (parsed) {
      const normalized = parsed.format('E.164');
      const found = contactCache.get(normalized);
      if (found) return found;
    }
  } catch {
    // Not a valid phone number
  }

  // Try stripping non-digits for phone matching
  const digitsOnly = handle.replace(/\D/g, '');
  if (digitsOnly.length >= 10) {
    const found = contactCache.get(digitsOnly);
    if (found) return found;

    // Try with +1 prefix
    const withPrefix = '+1' + digitsOnly.slice(-10);
    const foundWithPrefix = contactCache.get(withPrefix);
    if (foundWithPrefix) return foundWithPrefix;
  }

  return null;
}

export async function refreshContacts(): Promise<void> {
  await buildContactCache();
}

export function isContactsCacheBuilt(): boolean {
  return cacheBuilt;
}
```

Key implementation notes:
- Permission service uses node-mac-permissions for FDA detection
- Contact service builds lookup cache indexed by E.164 phone and lowercase email
- Handles various phone number formats via libphonenumber-js
- Cache is lazy-loaded and can be refreshed

```typescript
import { useState } from 'react';

interface PermissionOnboardingProps {
  onRetry: () => void;
}

export function PermissionOnboarding({ onRetry }: PermissionOnboardingProps) {
  const [opening, setOpening] = useState(false);

  const handleOpenSettings = async () => {
    setOpening(true);
    await window.electron.requestFullDiskAccess();
    // Give user time to toggle, then they'll click retry
    setOpening(false);
  };

  return (
    <div className="flex-1 flex items-center justify-center p-8">
      <div className="glass-panel max-w-md p-8 text-center">
        <div className="text-5xl mb-4">🔐</div>
        <h2 className="text-2xl font-semibold text-white mb-3">
          Full Disk Access Required
        </h2>
        <p className="text-white/70 mb-6 leading-relaxed">
          To read your iMessages, phoebeOS needs Full Disk Access permission.
          Your messages stay on your device - we never upload or share them.
        </p>

        <div className="space-y-3">
          <button
            onClick={handleOpenSettings}
            disabled={opening}
            className="w-full bg-blue-500 hover:bg-blue-600 disabled:bg-blue-500/50 text-white px-6 py-3 rounded-xl font-medium transition-colors"
          >
            {opening ? 'Opening Settings...' : 'Open System Settings'}
          </button>

          <button
            onClick={onRetry}
            className="w-full bg-white/10 hover:bg-white/20 text-white px-6 py-3 rounded-xl font-medium transition-colors"
          >
            I've Granted Access
          </button>
        </div>

        <div className="mt-6 pt-6 border-t border-white/10">
          <p className="text-sm text-white/50">
            In System Settings, find phoebeOS and toggle the switch to enable access.
          </p>
        </div>
      </div>
    </div>
  );
}
```
